class CSVIsleyici:
    def __init__(self):
        self.veriler = []
        self.basliklar = []
    
    def veri_ekle(self, basliklar, satirlar):
        self.basliklar = basliklar
        self.veriler = satirlar
    
    def goster(self):
        # Ba≈ülƒ±klarƒ± yazdƒ±r
        print(" | ".join(self.basliklar))
        print("-" * 50)
        # Verileri yazdƒ±r
        for satir in self.veriler:
            print(" | ".join(str(v) for v in satir))
    
    def filtrele(self, kolon_index, deger):
        return [s for s in self.veriler if s[kolon_index] == deger]

# Basit decorator
def zamanlayici(func):
    import time
    def wrapper(*args, **kwargs):
        baslangic = time.time()
        sonuc = func(*args, **kwargs)
        bitis = time.time()
        print(f"{func.__name__} √ßalƒ±≈üma s√ºresi: {bitis-baslangic:.4f} saniye")
        return sonuc
    return wrapper

@zamanlayici
def yavas_fonksiyon():
    toplam = 0
    for i in range(1000000):
        toplam += i
    return toplam

# Parametreli decorator
def tekrar(n):
    def decorator(func):
        def wrapper(*args, **kwargs):
            for _ in range(n):
                func(*args, **kwargs)
        return wrapper
    return decorator

@tekrar(3)
def selamla(isim):
    print(f"Merhaba {isim}!")

# Generator fonksiyon
def fibonacci_generator(n):
    a, b = 0, 1
    for _ in range(n):
        yield a
        a, b = b, a + b

# Bubble Sort
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

# Selection Sort
def selection_sort(arr):
    for i in range(len(arr)):
        min_idx = i
        for j in range(i+1, len(arr)):
            if arr[j] < arr[min_idx]:
                min_idx = j
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    return arr

# Quick Sort
def quick_sort(arr):
    if len(arr) <= 1:
        return arr
    pivot = arr[len(arr) // 2]
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    return quick_sort(left) + middle + quick_sort(right)

# Binary Search
def binary_search(arr, hedef):
    sol, sag = 0, len(arr) - 1
    while sol <= sag:
        orta = (sol + sag) // 2
        if arr[orta] == hedef:
            return orta
        elif arr[orta] < hedef:
            sol = orta + 1
        else:
            sag = orta - 1
    return -1

# Linear Search
def linear_search(arr, hedef):
    for i, eleman in enumerate(arr):
        if eleman == hedef:
            return i
    return -1

class Stack:

        self.items = []
    
    def push(self, item):
        self.items.append(item)
    
    def pop(self):
        if not self.is_empty():

    def peek(self):
        if not self.is_empty():
            return self.items[-1]
    
    def is_empty(self):
        return len(self.items) == 0

class Queue:
    def __init__(self):
        self.items = []
    
    def enqueue(self, item):
        self.items.insert(0, item)
    
    def dequeue(self):
        if not self.is_empty():
            return self.items.pop()
    
    def is_empty(self):
        return len(self.items) == 0

class API:

        self.id_counter = 1
    
    def get_all(self):

        return self.data[id]

            self.data[id].update(item)
            return self.data[id]
        return {"error": "Not found"}

        if id in self.data:
            del self.data[id]
            return {"message": "Deleted"}
        return {"error": "Not found"}
